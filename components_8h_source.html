272inline TemporalGeometry rect(const Point &p1, const Point &p2) {
273 Transform origin = identity();
274 for (int i = 0; i < 2; i++) {
275 origin.pos(i) = (p1.pos(i) + p2.pos(i)) / 2;
276 }
277 double width = std::abs(p1.pos(0) - p2.pos(0));
278 double height = std::abs(p1.pos(0) - p2.pos(0));
279 return TemporalGeometry{GeometryType::plane,
280 {origin.pos(0), origin.pos(1), origin.pos(2),
281 origin.rot(0), origin.rot(1), origin.rot(2), width,
282 height}};
