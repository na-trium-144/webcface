283inline TemporalGeometry rect(const Point &p1, const Point &p2) {
284 Transform origin = identity();
285 for (int i = 0; i < 2; i++) {
286 origin.pos(i) = (p1.pos(i) + p2.pos(i)) / 2;
287 }
288 double width = std::abs(p1.pos(0) - p2.pos(0));
289 double height = std::abs(p1.pos(0) - p2.pos(0));
290 return TemporalGeometry{GeometryType::plane,
291 {origin.pos(0), origin.pos(1), origin.pos(2),
292 origin.rot(0), origin.rot(1), origin.rot(2), width,
