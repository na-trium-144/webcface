227struct Polygon {
228 const Geometry &base;
229 explicit Polygon(const Geometry &rg) : base(rg) {
230 if (base.properties.size() % 3 != 0 || size() == 0) {
231 throw std::invalid_argument("number of properties does not match");
232 }
233 }
234 std::size_t size() const { return base.properties.size() / 3; }
235 Point at(std::size_t i) const {
236 if (i >= size()) {
237 throw std::out_of_range("Polygon::at(" + std::to_string(i) +
238 "), size = " + std::to_string(size()));
239 }
240 return Point{base.properties[i * 3 + 0], base.properties[i * 3 + 1],
241 base.properties[i * 3 + 2]};
242 }
243 Point operator[](std::size_t i) const { return at(i); }
244};
245inline TemporalGeometry polygon(const std::vector<Point> &points) {
246 std::vector<double> properties;
247 properties.reserve(points.size() * 3);
248 for (const auto &p : points) {
249 properties.push_back(p.pos(0));
250 properties.push_back(p.pos(1));
251 properties.push_back(p.pos(2));
252 }
253 return TemporalGeometry(GeometryType::polygon, std::move(properties));
254}

295inline TemporalGeometry rect(const Point &p1, const Point &p2) {
296 Transform origin = identity();
297 for (int i = 0; i < 2; i++) {
298 origin.pos(i) = (p1.pos(i) + p2.pos(i)) / 2;
299 }
300 double width = std::abs(p1.pos(0) - p2.pos(0));
301 double height = std::abs(p1.pos(0) - p2.pos(0));
302 return TemporalGeometry{GeometryType::plane,
303 {origin.pos(0), origin.pos(1), origin.pos(2),
304 origin.rot(0), origin.rot(1), origin.rot(2), width,
305 height}};
306}
