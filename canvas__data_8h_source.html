758struct Polygon {
759 const Geometry &base;
760 explicit Polygon(const Geometry &rg) : base(rg) {
761 if (base.properties.size() % 3 != 0 || size() == 0) {
762 throw std::invalid_argument("number of properties does not match");
763 }
764 }
765 std::size_t size() const { return base.properties.size() / 3; }
766 Point at(std::size_t i) const {
767 if (i >= size()) {
768 throw std::out_of_range("Polygon::at(" + std::to_string(i) +
769 "), size = " + std::to_string(size()));
770 }
771 return Point{base.properties[i * 3 + 0], base.properties[i * 3 + 1],
772 base.properties[i * 3 + 2]};
773 }
774 Point operator[](std::size_t i) const { return at(i); }
775};
776inline TemporalGeometry polygon(const std::vector<Point> &points) {
777 std::vector<double> properties;

779 for (const auto &p : points) {

781 properties.push_back(p.pos(1));
782 properties.push_back(p.pos(2));
783 }
784 return TemporalGeometry(GeometryType::polygon, std::move(properties));
