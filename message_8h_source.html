511struct WEBCFACE_DLL Log : public MessageBase<MessageKind::log> {
512 unsigned int member_id = 0;
513 std::shared_ptr<std::deque<LogLine>> log;
514 Log() = default;
515 Log(unsigned int member_id, const std::shared_ptr<std::deque<LogLine>> &log)
516 : member_id(member_id), log(log) {}
517 template <typename It>
518 Log(const It &begin, const It &end) : member_id(0) {
519 this->log = std::make_shared<std::deque<LogLine>>();
520 for (auto it = begin; it < end; it++) {
521 if constexpr (std::is_same_v<decltype(*it), LogLine>) {
522 this->log->push_back(*it);
523 } else {
524 this->log->emplace_back(it->toMessage());
525 }
526 }
527 }
528 explicit Log(const LogLine &ll) : member_id(0) {
529 this->log = std::make_shared<std::deque<LogLine>>(1);
530 this->log->front() = ll;
531 }
532 MSGPACK_DEFINE_MAP(MSGPACK_NVP("m", member_id), MSGPACK_NVP("l", log))
533};
