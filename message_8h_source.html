498struct WEBCFACE_DLL Log : public MessageBase<MessageKind::log> {
499 unsigned int member_id = 0;
500 std::shared_ptr<std::deque<LogLine>> log;
501 Log() = default;
502 Log(unsigned int member_id, const std::shared_ptr<std::deque<LogLine>> &log)
503 : member_id(member_id), log(log) {}
504 template <typename It>
505 Log(const It &begin, const It &end) : member_id(0) {
506 this->log = std::make_shared<std::deque<LogLine>>();
507 for (auto it = begin; it < end; it++) {
508 if constexpr (std::is_same_v<decltype(*it), LogLine>) {
509 this->log->push_back(*it);
510 } else {
511 this->log->emplace_back(it->toMessage());
512 }
513 }
514 }
515 explicit Log(const LogLine &ll) : member_id(0) {
516 this->log = std::make_shared<std::deque<LogLine>>(1);
517 this->log->front() = ll;
518 }
519 MSGPACK_DEFINE_MAP(MSGPACK_NVP("m", member_id), MSGPACK_NVP("l", log))
520};
