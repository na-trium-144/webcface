503struct Log : public MessageBase<MessageKind::log> {
504 unsigned int member_id = 0;
505 std::shared_ptr<std::deque<LogLine>> log;
506 Log() = default;
507 Log(unsigned int member_id, const std::shared_ptr<std::deque<LogLine>> &log)
508 : member_id(member_id), log(log) {}
509 template <typename It>
510 Log(const It &begin, const It &end) : member_id(0) {
511 this->log = std::make_shared<std::deque<LogLine>>();
512 for (auto it = begin; it < end; it++) {
513 if constexpr (std::is_same_v<decltype(*it), LogLine>) {
514 this->log->push_back(*it);
515 } else {
516 this->log->emplace_back(it->toMessage());
517 }
518 }
519 }
520 explicit Log(const LogLine &ll) : member_id(0) {
521 this->log = std::make_shared<std::deque<LogLine>>(1);
522 this->log->front() = ll;
523 }
524 MSGPACK_DEFINE_MAP(MSGPACK_NVP("m", member_id), MSGPACK_NVP("l", log))
